#include <Arduino.h>
#include <WiFi.h>
#include <WebServer.h>

// ================= WiFi НАСТРОЙКИ =================
const char* ssid = "ESP32-MIC";
const char* password = "12345678";

WebServer server(80);

// ================= НАСТРОЙКИ МИКРОФОНА =================
#define MIC_PIN 14

// ================= СТРИМ АУДИО =================
void handleAudio() {
  WiFiClient client = server.client();
  
  // HTTP заголовок для аудио потока
  String headers = "HTTP/1.1 200 OK\r\n";
  headers += "Content-Type: audio/wav\r\n";
  headers += "Access-Control-Allow-Origin: *\r\n";
  headers += "Cache-Control: no-cache\r\n";
  headers += "Connection: close\r\n\r\n";
  
  client.print(headers);
  
  // Отправляем WAV заголовок
  uint8_t wavHeader[44] = {
    0x52, 0x49, 0x46, 0x46,  // "RIFF"
    0xFF, 0xFF, 0xFF, 0xFF,  // размер файла
    0x57, 0x41, 0x56, 0x45,  // "WAVE"
    0x66, 0x6D, 0x74, 0x20,  // "fmt "
    0x10, 0x00, 0x00, 0x00,  // размер fmt (16)
    0x01, 0x00,              // формат PCM
    0x01, 0x00,              // моно
    0x40, 0x1F, 0x00, 0x00,  // частота 8000 Гц
    0x80, 0x3E, 0x00, 0x00,  // байт в секунду
    0x01, 0x00,              // выравнивание
    0x08, 0x00,              // бит на семпл (8 бит)
    0x64, 0x61, 0x74, 0x61,  // "data"
    0xFF, 0xFF, 0xFF, 0xFF   // размер данных
  };
  
  client.write(wavHeader, 44);
  
  // Бесконечный поток аудио
  unsigned long lastSample = micros();
  
  while (client.connected()) {
    unsigned long now = micros();
    
    // Отправляем семпл каждые 125 мкс (8000 Гц)
    if (now - lastSample >= 125) {
      int raw = analogRead(MIC_PIN);
      uint8_t sample = map(raw, 0, 4095, 0, 255);
      client.write(sample);
      lastSample = now;
    }
  }
}

void setup() {
  Serial.begin(115200);
  delay(1000);
  
  Serial.println("\n=== ПРОСТОЙ АУДИО СТРИМЕР ===");
  
  // Настройка микрофона
  analogReadResolution(12);
  analogSetAttenuation(ADC_11db);
  
  Serial.println("✅ Микрофон готов");
  
  // WiFi точка доступа
  WiFi.mode(WIFI_AP);
  WiFi.softAP(ssid, password);
  
  IPAddress ip = WiFi.softAPIP();
  Serial.println("\n📱 WiFi создан:");
  Serial.println("  SSID: ESP32-MIC");
  Serial.println("  Пароль: 12345678");
  Serial.print("  IP адрес: ");
  Serial.println(ip);
  
  // Маршруты
  server.on("/", handleRoot);
  server.on("/audio", handleAudio);
  
  server.begin();
  Serial.println("\n✅ Сервер запущен!");
  Serial.print("👉 Откройте в телефоне: http://");
  Serial.println(ip);
}

void loop() {
  server.handleClient();
}
