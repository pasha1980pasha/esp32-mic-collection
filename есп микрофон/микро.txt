// ================= НАСТРОЙКИ HW-484 (АНАЛОГОВЫЙ) =================
#define MIC_PIN 14  // GPIO14 (ADC1_CH6)
#define SAMPLE_WINDOW 50  // мс для измерения
#define SAMPLES 256       // количество семплов

// Переменные
int mic_value = 0;
int mic_max = 0;
int mic_min = 4095;
float mic_volume = 0;
bool mic_enabled = true;

// ================= ИНИЦИАЛИЗАЦИЯ АЦП =================
void init_adc_mic() {
  Serial.println("Инициализация АЦП для HW-484...");
  
  // Настройка АЦП
  analogReadResolution(12);  // 12 бит (0-4095)
  analogSetAttenuation(ADC_11db);  // Диапазон ~0-3.3V
  
  // Калибровка (измерение шума)
  long sum = 0;
  for (int i = 0; i < 100; i++) {
    sum += analogRead(MIC_PIN);
    delay(1);
  }
  int noise_level = sum / 100;
  
  Serial.printf("Уровень шума: %d\n", noise_level);
  Serial.println("✅ HW-484 (аналоговый) готов");
}

// ================= ЧТЕНИЕ МИКРОФОНА =================
void read_analog_mic() {
  if (!mic_enabled) return;
  
  unsigned long start = millis();
  int signal_max = 0;
  int signal_min = 4095;
  int sample_count = 0;
  
  // Сбор семплов в течение SAMPLE_WINDOW мс
  while (millis() - start < SAMPLE_WINDOW) {
    int sample = analogRead(MIC_PIN);
    
    if (sample > signal_max) signal_max = sample;
    if (sample < signal_min) signal_min = sample;
    
    sample_count++;
  }
  
  // Расчет peak-to-peak амплитуды
  int peak_to_peak = signal_max - signal_min;
  
  // Сохранение значений
  mic_value = analogRead(MIC_PIN);  // текущее значение
  mic_max = max(mic_max, signal_max);
  mic_min = min(mic_min, signal_min);
  
  // Конвертация в "громкость" (0.0 - 1.0)
  mic_volume = (float)peak_to_peak / 4095.0;
  
  // Детектор звука
  static bool sound_detected = false;
  if (peak_to_peak > 100) {  // Порог
    if (!sound_detected) {
      Serial.println("🔊 ЗВУК ОБНАРУЖЕН!");
      sound_detected = true;
    }
  } else {
    sound_detected = false;
  }
}

void handleData() {
  String json = "{";
  json += "\"current\":" + String(mic_value) + ",";
  json += "\"max\":" + String(mic_max) + ",";
  json += "\"min\":" + String(mic_min) + ",";
  json += "\"volume\":" + String(mic_volume, 3) + ",";
  json += "\"enabled\":" + String(mic_enabled ? "true" : "false");
  json += "}";
  
  server.send(200, "application/json", json);
}

void handleMicControl() {
  if (server.hasArg("enabled")) {
    mic_enabled = (server.arg("enabled") == "1");
    server.send(200, "text/plain", mic_enabled ? "ON" : "OFF");
  }
}

void handleCalibrate() {
  // Калибровка - измерение фонового шума
  mic_max = 0;
  mic_min = 4095;
  server.send(200, "text/plain", "Calibrated");
}

void setup() {
  Serial.begin(115200);
  delay(3000);
  
  Serial.println("\n=== HW-484 АНАЛОГОВЫЙ МИКРОФОН ===");
  Serial.println("Пин OUT → GPIO14 (ADC1_CH6)");
  Serial.println("Пин VCC → 3.3V");
  Serial.println("Пин GND → GND");
  Serial.println("Пин AO  → Не используется");
  
  // Инициализация АЦП
  init_adc_mic();
  
  // WiFi точка доступа
  WiFi.mode(WIFI_AP);
  WiFi.softAP(ssid, password);
  
  Serial.print("📶 WiFi: ");
  Serial.println(ssid);
  Serial.print("🌐 IP: ");
  Serial.println(WiFi.softAPIP());
  
  // Веб-сервер
  server.on("/", handleRoot);
  server.on("/data", handleData);
  server.on("/mic", HTTP_POST, handleMicControl);
  server.on("/calibrate", HTTP_POST, handleCalibrate);
  server.begin();
  
  Serial.println("✅ Сервер запущен");
  Serial.println("Откройте: http://" + WiFi.softAPIP().toString());
}

void loop() {
  server.handleClient();
  read_analog_mic();
  
  // Периодический вывод в Serial
  static unsigned long last_print = 0;
  if (millis() - last_print > 1000) {
    last_print = millis();
    Serial.printf("Значение: %4d | Громкость: %.3f\n", mic_value, mic_volume);
  }
  
  delay(10);
}
