#define BUFFER_SIZE 512

// ================= РАСПОЗНАВАНИЕ ГОЛОСА (простое) =================
String last_command = "";

// ================= ПРОСТОЙ ДЕТЕКТОР КОМАНД =================
String detect_simple_command(int16_t* audio, int length) {
  // 1. Расчет средней амплитуды
  int32_t sum_amplitude = 0;
  for (int i = 0; i < length; i++) {
    sum_amplitude += abs(audio[i]);
  }
  int avg_amplitude = sum_amplitude / length;
  
  // 2. Расчет пересечений нуля (частота)
  int zero_crossings = 0;
  for (int i = 1; i < length; i++) {
    if ((audio[i-1] < 0 && audio[i] >= 0) || 
        (audio[i-1] >= 0 && audio[i] < 0)) {
      zero_crossings++;
    }
  }
  float zero_cross_rate = (float)zero_crossings / length * 100;
  
  // 3. Детекция тишины/пауз
  int silent_count = 0;
  for (int i = 0; i < length; i++) {
    if (abs(audio[i]) < 100) silent_count++;
  }
  float silent_ratio = (float)silent_count / length * 100;
  
  // 4. Классификация команд
  Serial.printf("Детекция: Амплитуда=%d, Частота=%.1f%%, Тишина=%.1f%%\n",
                avg_amplitude, zero_cross_rate, silent_ratio);
  
  // Команда "СТОП" - громко, низкая частота, мало тишины
  if (avg_amplitude > 500 && zero_cross_rate < 12 && silent_ratio < 20) {
    command_confidence = map(avg_amplitude, 500, 1000, 0.6, 0.9);
    return "СТОП";
  }
  
  // Команда "ВКЛЮЧИ" - средняя громкость, средняя частота
  if (avg_amplitude > 200 && avg_amplitude < 500 && 
      zero_cross_rate > 8 && zero_cross_rate < 20) {
    command_confidence = map(avg_amplitude, 200, 500, 0.5, 0.8);
    return "ВКЛЮЧИ";
  }
  
  // Команда "ВЫКЛЮЧИ" - похожа на ВКЛЮЧИ, но короче
  if (avg_amplitude > 250 && avg_amplitude < 450 && 
      zero_cross_rate > 10 && zero_cross_rate < 18 &&
      silent_ratio > 30) {
    command_confidence = map(avg_amplitude, 250, 450, 0.5, 0.75);
    return "ВЫКЛЮЧИ";
  }
  
  // Команда "ПРИВЕТ" - высокая частота, средняя громкость
  if (avg_amplitude > 150 && zero_cross_rate > 15) {
    command_confidence = map(zero_cross_rate, 15, 25, 0.6, 0.8);
    return "ПРИВЕТ";
  }
  
  // Шум или нераспознано
  if (avg_amplitude > 100) {
    command_confidence = 0.3;
    return "ШУМ";
  }
  
  command_confidence = 0.0;
  return "ТИШИНА";
}

// ================= ЗАПИСЬ АУДИО С HW-484 =================
void record_audio() {
  static unsigned long next_sample_time = 0;
  static int buffer_pos = 0;
  
  if (!is_recording) return;
  
  unsigned long now = micros();
  if (now >= next_sample_time) {
    // Чтение с АЦП
    int raw = analogRead(MIC_PIN);
    
    // Конвертация в 16-bit signed (-2048 to 2047)
    int16_t sample = (int16_t)(raw - 2048);
    
    // Сохранение в буфер
    audio_buffer[buffer_pos++] = sample;
    
    // Обновление статистики
    if (sample > sound_max) sound_max = sample;
    if (sample < sound_min) sound_min = sample;
    
    // Если буфер заполнен
    if (buffer_pos >= BUFFER_SIZE) {
      buffer_pos = 0;
      
      // Проверяем время записи (максимум 3 секунды)
      if (millis() - record_start_time > 3000) {
        is_recording = false;
        Serial.println("⏹️  Запись завершена (3 сек)");
        
        // Распознаем команду
        last_command = detect_simple_command(audio_buffer, BUFFER_SIZE);
        Serial.printf("🎯 Распознано: %s (%.1f%%)\n", 
                      last_command.c_str(), command_confidence * 100);
      }
    }
    
    next_sample_time = now + (1000000 / SAMPLE_RATE);
  }
}

// ================= API =================
void handleData() {
  int current_level = analogRead(MIC_PIN);
  
  String json = "{";
  json += "\"command\":\"" + last_command + "\",";
  json += "\"confidence\":" + String(command_confidence, 2) + ",";
  json += "\"vr_enabled\":" + String(voice_recognition_enabled ? "true" : "false") + ",";
  json += "\"is_recording\":" + String(is_recording ? "true" : "false") + ",";
  json += "\"current_level\":" + String(current_level) + ",";
  json += "\"max_level\":" + String(sound_max + 2048) + ",";
  json += "\"min_level\":" + String(sound_min + 2048) + ",";
  json += "\"avg_level\":" + String((sound_max + sound_min) / 2.0 + 2048, 1) + ",";
  json += "\"free_memory\":" + String(ESP.getFreeHeap()) + ",";
  json += "\"uptime\":" + String(millis());
  json += "}";
  
  server.send(200, "application/json", json);
}

void handleRecord() {
  if (server.hasArg("action")) {
    String action = server.arg("action");
    
    if (action == "start") {
      is_recording = true;
      record_start_time = millis();
      sound_max = -2048;
      sound_min = 2047;
      memset(audio_buffer, 0, sizeof(audio_buffer));
      Serial.println("🎤 НАЧАЛО ЗАПИСИ... Говорите команду!");
      server.send(200, "text/plain", "Recording started");
    } else if (action == "stop") {
      is_recording = false;
      last_command = detect_simple_command(audio_buffer, BUFFER_SIZE);
      Serial.printf("⏹️  ЗАПИСЬ ОСТАНОВЛЕНА | Команда: %s (%.1f%%)\n",
                    last_command.c_str(), command_confidence * 100);
      server.send(200, "text/plain", "Recording stopped");
    }
  }
}

void handleVoiceControl() {
  if (server.hasArg("enabled")) {
    voice_recognition_enabled = (server.arg("enabled") == "1");
    Serial.println(voice_recognition_enabled ? "✅ Распознавание ВКЛ" : "⏸️ Распознавание ВЫКЛ");
    server.send(200, "text/plain", voice_recognition_enabled ? "VR ON" : "VR OFF");
  }
}

void loop() {
  server.handleClient();
  
  // Запись аудио
  record_audio();
  
  // Автоматический детектор голоса
  static unsigned long last_auto_record = 0;
  if (voice_recognition_enabled && !is_recording && 
      millis() - last_auto_record > 5000) {
    
    int current = analogRead(MIC_PIN);
    int level = abs(current - 2048);
    
    // Если громкий звук - начинаем запись автоматически
    if (level > 300) {
      is_recording = true;
      record_start_time = millis();
      sound_max = -2048;
      sound_min = 2047;
      memset(audio_buffer, 0, sizeof(audio_buffer));
      Serial.println("🔊 АВТОЗАПИСЬ! Обнаружен звук...");
      last_auto_record = millis();
    }
  }
  
  // Периодический вывод статуса
  static unsigned long last_status = 0;
  if (millis() - last_status > 2000) {
    last_status = millis();
    
    int mic_val = analogRead(MIC_PIN);
    int level = abs(mic_val - 2048);
    
    Serial.printf("Микрофон: %4d | Уровень: %4d | Команда: %s",
                  mic_val, level, last_command.c_str());
    
    if (is_recording) {
      Serial.printf(" | Запись: %d мс", millis() - record_start_time);
    }
    Serial.println();
  }
  
  delay(1);
}
